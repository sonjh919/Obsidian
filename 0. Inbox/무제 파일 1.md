import redis  
import json  
import boto3# Redis 연결 설정  
redis_host_topic = "172.31.46.212"  
redis_port_topic = 6379  
sqs = boto3.client('sqs')  
redis_client = redis.StrictRedis(host=redis_host_topic, port=redis_port_topic)def lambda_handler(event, context):  
    # SQS 메시지 수신  
    message_body = event['Records'][0]['body']    # 레디스에 메시지 publish  
    channel = "user"  
    redis_client.publish(channel, message_body)    print(message_body)        # print("Published message to Redis:", message)    return {  
        'statusCode': 200,  
        'body': json.dumps('Messages published to Redis successfully')  
    }
